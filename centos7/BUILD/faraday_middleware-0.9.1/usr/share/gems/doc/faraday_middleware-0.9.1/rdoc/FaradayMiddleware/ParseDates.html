def parse_dates!(value)
  case value
  when Hash
    value.each do |key, element|
      value[key] = parse_dates!(element)
    end
  when Array
    value.each_with_index do |element, index|
      value[index] = parse_dates!(element)
    end
  when @regexp
    Time.parse(value)
  else
    value
  end
end
