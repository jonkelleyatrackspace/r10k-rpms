  entries = []
  if @file.file?
    entries = @file.readlines.map(&:chomp)
  end
  entries
end

def write(entries)
  if ! @file.parent.directory?
    raise R10K::Git::GitError, "Cannot write #{@file}; parent directory does not exist"
  end
  @file.open("w") do |fh|
    entries.each do |entry|
      fh.puts(entry)
    end
  end
end
