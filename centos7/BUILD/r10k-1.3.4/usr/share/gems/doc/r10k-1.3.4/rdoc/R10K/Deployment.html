def load_environments
  @_environments = []
  sources.each do |source|
    @_environments += source.environments
  end
end

def load_sources
  sources = @config.setting(:sources)
  @_sources = sources.map do |(name, hash)|
    R10K::Source.from_hash(name, hash)
  end
end
