  def checkout(ref)
    if ref.resolvable?
      git ["checkout", "--force", @ref.sha1], :path => @full_path
    else
      raise R10K::Git::UnresolvableRefError.new(
        "Cannot check out unresolvable ref '#{@ref}'",
        :git_dir => @full_path)
    end
  end
