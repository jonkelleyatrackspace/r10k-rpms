    def call
      logger.notice "Deploying environment #{@environment.dirname}"
      @environment.sync

      if @update_puppetfile
        task = R10K::Task::Puppetfile::Sync.new(@environment.puppetfile)
        task_runner.insert_task_after(self, task)
      end
    end
