    def self.command
      @cmd ||= Cri::Command.define do
        name    'deploy'
        usage   'deploy <subcommand>'
        summary 'Puppet dynamic environment deployment'

        description "`r10k deploy` implements the Git branch to Puppet environment workflow\n(https://puppetlabs.com/blog/git-workflow-and-puppet-environments/).\n"

        required :c, :config, 'Specify a configuration file'

        run do |opts, args, cmd|
          puts cmd.help(:verbose => opts[:verbose])
          exit 0
        end
      end
    end
