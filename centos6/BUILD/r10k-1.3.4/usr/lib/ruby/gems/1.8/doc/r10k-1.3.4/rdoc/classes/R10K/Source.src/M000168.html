    def self.from_hash(name, hash)
      hash.extend R10K::Util::CoreExt::HashExt::SymbolizeKeys
      hash.symbolize_keys!

      basedir = hash.delete(:basedir)

      type = hash.delete(:type)
      type = type.is_a?(String) ? type.to_sym : type

      generate(type, name, basedir, hash)
    end
