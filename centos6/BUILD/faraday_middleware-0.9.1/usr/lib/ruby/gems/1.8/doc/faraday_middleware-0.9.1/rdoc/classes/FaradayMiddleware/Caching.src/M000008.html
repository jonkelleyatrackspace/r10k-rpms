    def call(env)
      if :get == env[:method]
        if env[:parallel_manager]
          # callback mode
          cache_on_complete(env)
        else
          # synchronous mode
          response = cache.fetch(cache_key(env)) { @app.call(env) }
          finalize_response(response, env)
        end
      else
        @app.call(env)
      end
    end
