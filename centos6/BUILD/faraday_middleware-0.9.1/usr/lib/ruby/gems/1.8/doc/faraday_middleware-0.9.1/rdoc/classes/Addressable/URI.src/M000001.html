    def normalized_query
      fresh = @normalized_query.nil?
      query = normalized_query_without_ordering_fix
      if query && fresh
        @normalized_query = query.split('&', -1).sort_by {|q| q[0..(q.index('=')||-1)] }.join('&')
      else
        query
      end
    end
