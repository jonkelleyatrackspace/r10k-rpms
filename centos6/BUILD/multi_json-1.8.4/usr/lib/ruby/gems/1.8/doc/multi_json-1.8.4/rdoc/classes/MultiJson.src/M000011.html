  def load_adapter(new_adapter)
    case new_adapter
    when String, Symbol
      load_adapter_from_string_name new_adapter.to_s
    when NilClass, FalseClass
      load_adapter default_adapter
    when Class, Module
      new_adapter
    else
      raise ::LoadError
    end
  rescue ::LoadError
    raise ArgumentError, 'Did not recognize your adapter specification.'
  end
