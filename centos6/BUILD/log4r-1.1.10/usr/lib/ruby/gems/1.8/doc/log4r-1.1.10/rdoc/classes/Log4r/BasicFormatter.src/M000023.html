    def format(event)
      buff = sprintf(@@basicformat, MaxLevelLength, LNAMES[event.level],
             event.name)
      buff << (event.tracer.nil? ? "" : "(#{event.tracer[0]})") + ": "
      buff << format_object(event.data) + "\n"
      buff
    end
